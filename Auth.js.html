/** 이메일 주소와 비밀번호로 사용자 로그인
 * @class Auth
 * @function _signin_password
 * @param {string} email 유저 이메일
 * @param {string} password 유저 패스워드
 * @returns {string} 유저 uid
 */
export const _signin_password = async (email, password) => {
  let uid = "";
  await signInWithEmailAndPassword(auth, email, password)
    .then(async (userCredential) => {
      // Signed in
      const user = userCredential.user;
      uid = user.uid;
      // ...
    })
    .catch(async (error) => {
      const errorCode = error.code;
      const errorMessage = error.message;

      // 에러메세지 출력
      console.log(errorMessage);
    });

  return uid;
};

/** 구글 로그인
 * @class Auth
 * @function _signin_google
 */
export const _signin_google = async () => {
  // Retrieve Google provider object
  const provider = new firebase.auth.GoogleAuthProvider();
  // Set language to the default browser preference
  firebase.auth().useDeviceLanguage();
  // Start sign in process

  try {
    await firebase.auth().signInWithPopup(provider);
  } catch (error) {
    console.log(error.message);
  }
};

/** 구글 로그인
 * @function _signin_password
 * @class Auth
 */
export const sign_out = async () => {
  try {
    await firebase.auth().signOut();
    console.log("로그아웃");
  } catch (error) {
    console.log(error.message);
  }
};
